Correlations between measures of autistic traits and social anxiety

All measures of social anxiety showed moderate-to-strong correlations between each other and with AQ. We used Spearman’s rank correlations as questionnaire scores did not follow a normal distribution (except for SIAS).

# functions for plotting
# ~~~~~~~~~~~~~~~~~~~~~~
# color-code correlation coefficients in the upper panel
upperFn <- function(data, mapping, method="spearman", use="pairwise", ...){
  
  # compute correlation coefficient
  x <- eval_data_col(data, mapping$x)
  y <- eval_data_col(data, mapping$y)
  corr <- cor(x, y, method=method, use=use)

  # calculate colour based on coefficient
  colFn <- colorRampPalette(brewer.pal(n=8, name='RdYlBu'), interpolate ='spline')
  fill <- colFn(100)[findInterval(corr, seq(-1, 1, length=100))]

  # print correlation coefficient
  p <- ggally_text(mapping=aes(), label=paste(as.character(round(corr,2))),
              xP=0.5, yP=0.5, color='black',...) +
    theme_bw() + theme(
      panel.grid=element_blank(),
      panel.border=element_rect(colour='white'),
      panel.background=element_rect(fill=fill, colour='white', size=11))
  p 
  }

# make the correlation plots in the lower panel pretty
lowerFn <- function(data, mapping, method='lm',...) {
  p <- ggplot(data=data, mapping=mapping) +
    geom_point(fill='steelblue4', size=0.5) +
    geom_smooth(method=method, size=1, fill='steelblue4', colour='steelblue4', alpha=0.25,...) +
    theme_bw() + theme(panel.grid=element_blank(),
                       panel.border=element_blank(), 
                       panel.background=element_rect(fill='grey92', colour='white'))
  p
}

# make the histograms in the diagonal pretty
diagFn <- function(data, mapping, method='barDiag',...) {
  p <- ggplot(data=data, mapping=mapping) +
    geom_histogram(colour='white', fill='steelblue4') +
    theme_bw() + theme(panel.grid=element_blank(), 
                       panel.border=element_blank())
  p
}

  cormat$rho[i] <- rcorr(d.quest[,mat$var1[i]], d.quest[,mat$var2[i]], type='spearman')$r[2]

Automatic imitation task

Data exclusion

Single-trial data were excluded from the reaction times (RT) and error rates (ER) analyses according to the preregistered criteria:

RT analyses

Trials were excluded if:

  • No response was made.
  • The response was incorrect.
  • RT were smaller than 200 ms or greater than 1000 ms.
  • RT deviated 1.5 times the inter-quartile range within each condition and participant.

ER analyses

Trials were excluded if:

  • No response was made.
# *******************************
# IDENTIFY TRIALS TO BE EXCLUDED 
# *******************************
# create data frame to mark trials to reject
ntrials <- nrow(d.task)
d.excl <- data.frame(id=d.task$id, trial=d.task$trial, missing=logical(ntrials), 
                   incorrect=logical(ntrials), outlier=logical(ntrials), 
                   slowrt=logical(ntrials))

# MISSING: trials in which no response was made
# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
d.excl$missing <- is.na(d.task$rt)

# INCORRECT: trials in which response was incorrect
# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
d.excl$incorrect <- d.task$error

# FAST/SLOW RT: trials with RT 200 ms or slower than 1000 ms
# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
d.excl$slowrt <- d.task$rt < 200 | d.task$rt > 1000

# OUTLIERS: trials with RT that deviate from 1.5*IQR within condition and participant
# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
# for each participant...
for(id in levels(d.task$id)){
  # for each condition
  for(cond in levels(d.task$condition)){
    
    # select data for that particular condition
    d.cond <- d.task[d.task$id==id & d.task$condition==cond,]
    
    # find outliers: RT outside 1.5*IQR
    outliers <- d.cond$rt %in% boxplot.stats(d.cond$rt)$out 
    d.excl$outlier[(d.excl$id %in% d.cond$id[outliers]) & (d.excl$trial %in% d.cond$trial[outliers])] <- TRUE
    
  }
}

# Trials to reject
# ~~~~~~~~~~~~~~~~~
# mark trials that need to be rejected for the RT and ER analyses:
# note that the total number of trials rejected is not the sum of all trials
# identified as missing/incorrect/outliers/etc. Some of the trials might
# have been rejected in more than one category (e.g. slow RT may
# be detected also as outliers)
d.task$excl_rt <- d.excl$missing | d.excl$incorrect | d.excl$slowrt | d.excl$outlier
d.task$excl_er <- d.excl$missing

# IDENTIFY PARTICIPANTS TO BE EXCLUDED BASED ON NUMBER OF TRIALS 
# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
excl.id <- d.task %>%
  
  # compute number of trials excluded per participant and condition
  group_by(id, condition) %>%
  dplyr::summarise(n_excl=sum(excl_rt)) %>%
  
  # exclude participants with more than 40% of the trials of at  
  # least one condition were rejected
  group_by(id) %>%
  dplyr::summarise(excl=any(n_excl>round(32*0.40))) 

# mark excluded participants in dataset
d.task$excl_id <- d.task$id %in% excl.id$id[excl.id$excl]


# PREPARE DATASET FOR ANOVA (aggregate data)
# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
# RT: exclude participants, missing data, incorrect trials, outliers
